async def create_rooms(guild):
    clear()
    print(f"{DARK_BLUE}{BOLD}═══ CREATE ROOMS ═══{RESET}\n")
    
    name = input(f"{DARK_BLUE}[?]{WHITE} Room Name: ").strip()
    count = int(input(f"{DARK_BLUE}[?]{WHITE} Count: ").strip())
    
    print(f"\n{DARK_BLUE}[+]{WHITE} Creating {count} rooms...\n")
    
    created = 0
    chunk = 50  # a-shell مناسب
    
    for i in range(0, count, chunk):
        batch = min(chunk, count - i)
        tasks = []
        for j in range(batch):
            tasks.append(guild.create_text_channel(name))
        results = await asyncio.gather(*tasks, return_exceptions=True)
        for r in results:
            if not isinstance(r, Exception):
                created += 1
                if created % 20 == 0:
                    print(f"{GREEN}[+]{WHITE} {created} rooms")
    
    print(f"\n{GREEN}[+]{WHITE} {created} rooms created!")
    input(f"\n{DARK_BLUE}[?]{WHITE} Enter...")
    await main_menu(guild)

async def give_admin(guild):

    members = [m for m in guild.members if not m.bot]
    if not members:
        print(f"{RED}[!]{WHITE} No members!")

        await main_menu(guild)
        return
    
    print(f"{GREEN}[+]{WHITE} {len(members)} members")
    confirm = input(f"{RED}[!]{WHITE} Give ADMIN? (y/n): ").strip().lower()

        print(f"{RED}[!]{WHITE} Cancelled!")
        input(f"{DARK_BLUE}[?]{WHITE} Enter...")
        await main_menu(guild)
        return
    
    admin_role = discord.utils.get(guild.roles, name="ADMIN")
    if not admin_role:
        try:
            admin_role = await guild.create_role(name="ADMIN", permissions=discord.Permissions.all())
            print(f"{GREEN}[+]{WHITE} Created ADMIN role")
        except:
            admin_role = guild.roles[-1] if guild.roles else None
    
    if not admin_role:
        print(f"{RED}[!]{WHITE} No role!")
        input(f"{DARK_BLUE}[?]{WHITE} Enter...")
        await main_menu(guild)
        return
    
    print(f"\n{DARK_BLUE}[+]{WHITE} Giving ADMIN...\n")
    
    given = 0
    chunk = 30
    
    for i in range(0, len(members), chunk):
        batch = members[i:i+chunk]
        tasks = [m.add_roles(admin_role) for m in batch]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        for idx, r in enumerate(results):
            if not isinstance(r, Exception):
                given += 1
                if given % 20 == 0:
                    print(f"{GREEN}[+]{WHITE} {given} members")
    
    print(f"\n{GREEN}[+]{WHITE} {given} members got ADMIN!")
    input(f"\n{DARK_BLUE}[?]{WHITE} Enter...")
    await main_menu(guild)
